(* Use this in Wolfram Mathematica *)

DivisorSum[n_] := Total[Divisors[n]] - n
AmicableQ[n_] := DivisorSum[n] != n && DivisorSum[DivisorSum[n]] == n
Total[Select[Range[9999], AmicableQ]]
